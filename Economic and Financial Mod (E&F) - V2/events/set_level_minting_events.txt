namespace = vo_tax_income_add

vo_tax_income_add.1 = { 
	type = country_event
	hidden = yes
	immediate = {
		if = {
			limit = {
				has_variable = tax_income_rate_add
			}
			change_variable = {
				name = tax_income_rate_add
				add = 0.005
			}
		}
		else = {
			set_variable = {
				name = tax_income_rate_add
				value = 0.005
			}
		}
		remove_modifier = modifier_vo_tax_income_add
		add_modifier = {
			name = modifier_vo_tax_income_add
			multiplier = var:tax_income_rate_add
		}
	}
}
