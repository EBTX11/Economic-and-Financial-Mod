algerian_dinar_currency_law_list = {
	clear_variable_list = algerian_dinar_currency_law_list
	every_country = {
		limit = {
			has_law = law_type:law_algerian_dinar_currency
			NOT = {
				is_subject = yes 
				is_subject_type = subject_type_customs_union
				is_subject_type = subject_type_protectorate         
			}
		}
		root = {
			add_to_variable_list = {
				name = algerian_dinar_currency_law_list
				target = prev
			}
		}
	}
	root  = {
		if = {
			clear_variable_list = list_before_order
			every_in_list = {
				variable = algerian_dinar_currency_law_list
				root = {
					add_to_variable_list = {
						name = list_before_order
						target = prev
					}
				}
			}
			clear_variable_list = algerian_dinar_currency_law_list_ordered_1
			if = {
				ordered_in_list = {
					variable = list_before_order
					order_by = national_capacity
					max = 1
					check_range_bounds = no
					root = {
						add_to_variable_list = {
							name = algerian_dinar_currency_law_list_ordered_1
							target = prev
						}
					}
				}
			}
			every_in_list = {
				variable = algerian_dinar_currency_law_list_ordered_1
				if = {
					remove_modifier = algerian_dinar_reference
					add_modifier = {
						name = algerian_dinar_reference
					}
				}
			}
		}
		#clear_variable_list = algerian_dinar_currency_list
	}
