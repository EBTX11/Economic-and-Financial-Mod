#shares
ai_exchange_undervalued_shares_decisions_1 = {
	is_shown = {
		is_ai = yes
	}

	possible = {
		has_modifier = has_central_bank
		var:ai_exchange_decisions_time_shares_undervalued_1 = 0	
		not  = {
			has_law = law_type:law_shares_currency
			or = {
				is_subject = yes
				is_subject_type = subject_type_customs_union
				is_subject_type = subject_type_protectorate
				has_law = law_type:law_isolationism
			}
		}
	}          

	when_taken = {
		set_variable = {
			name = ai_exchange_decisions_time_shares_undervalued_1
			value = 1
		}

		every_country = {
			limit  = {
				is_player = yes
			}
			save_scope_as = is_player_for_notification
		}
	
		random_country = {
			limit = {
				has_law = law_type:law_shares_currency
				money_supply > money_supply_stockpile_by_other_country_indexed_prestige_to_buy 
				money_supply > 100000000
				has_modifier = has_central_bank
				not = {
					has_modifier = global_monetary_reference
					is_player = yes
					or = {
						is_subject_of = root 
						is_subject_type = subject_type_customs_union
						is_subject_type = subject_type_protectorate   
					} 
				}
			}
			save_scope_as = undervalued_shares_to_gold
		}

		root = {
			save_scope_as = gold_to_shares_seller
		}

		set_variable = {
			name = shares_quantity_var
			value =  { 5000000 15000000 } 
		}

		set_variable = {
			name = stockpiling_undervalued_shares_state_1_buy_in_gold
			value = var:shares_quantity_var
		}
		change_variable = {
			name = stockpiling_undervalued_shares_state_1_buy_in_gold
			multiplier = scope:undervalued_shares_to_gold.money_value_rapported_inflation
		}

		set_variable = {
			name = stockpiling_undervalued_shares_state_buy_vriablity
			value = var:shares_quantity_var
		}

		scope:gold_to_shares_seller = {
			if = {
				limit = {
					exists = scope:undervalued_shares_to_gold
				}
				every_scope_state = {
					if = {
						limit = {
							has_modifier = bank_central_historic_place
							is_capital = yes
						}
						save_scope_as = gold_to_shares_seller_capital_state
						#########

						change_variable = {
							name = central_bank_gold_reserves_state_1
							subtract = owner.var:stockpiling_undervalued_shares_state_1_buy_in_gold
						}
		
						#########
						change_variable = {
							name = stockpiling_shares_state_1
							add = owner.var:stockpiling_undervalued_shares_state_buy_vriablity
						}
					}
				}
				scope:undervalued_shares_to_gold = {
					every_scope_state = {
						if = {
							limit = {
								has_modifier = bank_central_historic_place
								is_capital = yes
							}
							#########

							change_variable = {
								name = central_bank_gold_reserves_state_1
								add = scope:gold_to_shares_seller.var:stockpiling_undervalued_shares_state_1_buy_in_gold
							}
		
							#########
							change_variable = {
								name = stockpiling_shares_state_1
								subtract = scope:gold_to_shares_seller.var:stockpiling_undervalued_shares_state_buy_vriablity
							}
						}
					}
				}
			}
		}
	}

	ai_chance = {
		base = 100
	}
}
