namespace = financial_center_chi_event

financial_center_chi_event.1 = {

	type = country_event
	placement = financial_center_state
	hidden = yes

	title = "Financial Center I"

	duration = 1

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_SHAOZHOU
			}
			save_scope_as = financial_center_state
		}
		scope:financial_center_state = {
			create_building = {
				building = building_financial_centre_chi
				level = 25
			}
		}
	}
}
