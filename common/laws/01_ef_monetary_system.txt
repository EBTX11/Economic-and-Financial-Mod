law_bimetallism_standard = {
	group = lawgroup_monetary_system
	institution = institution_economic_central_bank

	icon = "gfx/interface/icons/law_icons/bimetallism_standard.dds"

	unlocking_technologies = {
		metalique_standard
	}
	unlocking_laws = {
		law_silver_standard
		law_gold_standard
	}
	
	can_enact = {
		NOT = { 
			is_not_market_owner = yes
			has_law = law_type:law_isolationism
			has_modifier = central_bank_bankruptcy_country
			has_modifier = currency_crisis_country
			has_modifier = monetary_systeme_transition
		}
	} 

	on_activate = {
		activate_law = law_type:law_bimetallic_ratio_15 
		on_activate_monetary_system_law = yes  

		every_country = { 
			limit = { 
				this = root
				has_modifier = has_central_bank
				Not = {
					or = {
						market_owner_is_root_univ_with_buiding = yes
						is_subject_custom_trigger = yes
					}
				} 
			} 
			capital= {
				save_scope_as = capital_state
			} 
			save_scope_as = target_country

			every_country = {
				limit = {
					is_player = yes  
				}   
				trigger_event = 00_ef_economic_event.26
			}    
			change_global_variable = {
				name = global_silver_production_cumulative_production
				add = {
					value = global_silver_production
					multiply = 0.03
				}
			} 
		}   
	}

	progressiveness = 50
	 
	ai_will_do = {
		always = no
	}
}
