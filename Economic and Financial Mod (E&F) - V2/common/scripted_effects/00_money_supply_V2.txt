bank_central_destroy = {
    capital = {
        save_scope_as = capital_scope
    }
    every_scope_state = {
        if = {
            limit = {
                has_modifier = modifier_capital
            } 
            remove_modifier = modifier_capital
        }
    }
    scope:capital_scope = {
        if = {
            limit = {
                has_modifier = modifier_capital
            }
            remove_modifier = modifier_capital
        }
        add_modifier = {
            name = modifier_capital
            months = short_modifier_time
        }
    }
    every_scope_state = {
        limit = {
            NOT = {
                has_modifier = modifier_capital
            }
        }
        remove_building = building_bank
    }
}
