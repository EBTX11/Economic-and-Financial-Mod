	flavored_company = yes
	
	building_types = {  
		building_paper_mills
	}
	
	potential = {

	}
	
	attainable = {

			any_scope_state = {
				state_region = s:STATE_RIGA
			}
		}

			}
		}
	}
	
	prosperity_modifier = {
		building_paper_mills_throughput_add = 0.1 
		country_prestige_mult = 0.1
		state_radicals_from_sol_change_mult = -0.1
	}	
	
	ai_weight = {

	}
}
