		}
		change_variable = {
			name = circulating_franc_belgian_franc_c_var_1
			add = { 50000009 100000009 }
		}
		scope:central_bank_site = {
			change_variable = {
				name = stockpiling_franc_belgian_franc_state_1
				add  = {
					value = owner.var:circulating_franc_belgian_franc_c_var_1
					multiply = 0.1
				}
			}
			change_variable = {
				name = silver_state_1
				add = {
					value = owner.var:circulating_franc_belgian_franc_c_var_1
					add = var:stockpiling_franc_belgian_franc_state_1
					multiply = owner.var:money_value_target_1
				}
			}
			change_variable = {
				name = gold_state_1
				add = {
					value = var:silver_state_1
					multiply = silver_to_gold_rate
				}
			}
		}

			limit = {
