namespace = law_event_yuan

#yuan
law_event_yuan.1 = {
	type = country_event
	placement = root
	hidden = yes

	title = "Current circulation - yuan"

	duration = 1

	immediate = {                                     
		capital = {
			save_scope_as = central_bank_site
		}
		scope:central_bank_site = {
			if = {
				limit = {
					not = {
						root = {
							has_law = law_type:law_yuan_currency
						}
					}
				}
				change_variable = {
                    name = central_bank_gold_reserves_state_1
                    add = { 175000000 225000000 }
                }
				change_variable = {
					name = stockpiling_yuan_state_1
					add = 100000000
				}
			} 
		}
		if = {
			limit = {
				not = {
					root = {
						has_law = law_type:law_yuan_currency
					}
				}
			}
			set_variable = {
				name = circulating_yuan_c_var_1
				value = 100000000
			}
		}
		activate_law = law_type:law_yuan_currency
	}
}
