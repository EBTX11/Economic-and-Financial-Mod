## for adding national stockpile at game start
initialize_historic_stockpile = {
	add_technology_researched = stockpiling_goods
	add_technology_researched = national_stockpile

	$STOCKPILE_SITE$ = {
		create_building = {
			building = building_national_stockpile
			level = 1
		}
		add_modifier = {
			name = national_stockpile_place
		}
		add_modifier = {
			name = national_stockpile_historic_place #central_bank_historic_place
		}
		owner = {
			add_modifier = {
				name = has_national_stockpile
			}
			set_variable = {
				name = stockpile_location
				value = $STOCKPILE_SITE$
			}
		}
	}
}

macro_facilities_on_action_nc = {
	if = {
		limit = { ### check to make sure we're ready and stable enough
			has_technology_researched = national_stockpile
			is_revolutionary =  no
			or = {
				is_valid_country = yes
				country_already_national_stockpile = yes
				gdp_view >= 1
				is_player = yes
			}
		}
		macro_facilities_ns = {
			NATIO_STOCK_SITE = capital
			NATIO_BLDG_TYPE = building_national_stockpile
		}
	}
}
macro_facilities_ns = {
	if = {
		limit = { has_technology_researched = national_stockpile }
		$NATIO_STOCK_SITE$ = {
			create_building = {
				building = $NATIO_BLDG_TYPE$
				level    = 1
				reserves=1
			}
		}
	}
	national_stockpile_modifier = yes
}
