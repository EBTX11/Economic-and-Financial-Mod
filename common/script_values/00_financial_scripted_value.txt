country_credit_pondered_mean = {
	value = 0

	if = {
		limit = {
			global_country_ranking = 1
			not = {
				is_player = yes
			}
		}
		add = 10
	}

	if = {
		limit = {
			has_modifier = has_central_bank
			#global_country_ranking > 1
			or  = {
				and  = {
					global_country_ranking > 1
					global_country_ranking < 50
				}
				is_player = yes
			}
		}

		value = country_credit_score
		divide = 31
		add = institution_economic_central_bank_lvl

		#is_at_war
		if = {
			limit  = {
				is_at_war = yes
			}
			multiply = is_at_war_years_pulse_malus
		}
		#taking_loans
		if = {
			limit  = {
				taking_loans = yes
			}
			multiply = 0.75
		}
		#credit_rating_national_debt_compared_gdp
		if = {
			limit  = {
				country_credit_rating_national_debt_compared_gdp = 0
			}
			multiply = country_credit_rating_national_debt_compared_gdp_malus_100
		}
	}

	#<50 and no Bc
	if = {
		limit = {
			global_country_ranking < 50
			not = {
				has_modifier = has_central_bank
			}
		}
		add = 4
	}
	#>= 50
	if = {
		limit = {
			global_country_ranking >= 50
			global_country_ranking < 100
			not = {
				has_modifier = has_central_bank
			}
		}
		add = 3
	}
	#>= 100
	if = {
		limit = {
			global_country_ranking >= 100
			global_country_ranking < 150
			not = {
				has_modifier = has_central_bank
			}
		}
		add = 2
	}
	#>= 150
	if = {
		limit = {
			global_country_ranking >= 150
			not = {
				has_modifier = has_central_bank
			}
		}
		add = 1
	}
}

price_bond = {
	value = 0
	if = {
		limit = {
			player.var:bond_in_on = 0
		}
		value = player.var:bond_quantity
		multiply = price_bond_in_buyer_currency #bond_price_in_gold #bond_price
	}
	if = {
		limit = {
			player.var:bond_in_on = 1
		}
		value = price_bond_in_currency_converti
	}
}
price_bond_in_gold = {
	value = player.var:bond_quantity
	multiply = bond_price_in_gold #bond_price
}
gold_to_currency_rate = { #for gold
    value = 1
    divide = player.money_value_in_gold
}
price_bond_in_buyer_currency = { #for gold
	value = bond_price_in_gold #price_bond
	multiply = gold_to_currency_rate
}
price_bond_in_currency = { #for market owner currency
	value = player.var:bond_quantity
	multiply = bond_price
}
price_bond_in_currency_converti = { #for market owner currency
	value = price_bond_in_currency
	multiply = exchange_rate_for_international_trade_buy_for_target_market_panel
}
