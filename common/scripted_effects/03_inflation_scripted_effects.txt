currency_strength_modifier = {

	#strong_currency
	if = {
		limit = {
			has_modifier = reference_currency
		}
		remove_modifier = reference_currency
	}
	if = {
		limit = {
			has_modifier = has_central_bank
            has_modifier = global_monetary_reference
		}
		add_modifier = {
			name = reference_currency
			multiplier = difference_with_average_gold_exchange_rate_currencies_revers_100 
		}
	}
	#strong_currency
	if = {
		limit = {
			has_modifier = strong_currency
		}
		remove_modifier = strong_currency
	}
	if = {
		limit = {
			money_value_rapported_inflation > global_var:money_value_median
			has_modifier = has_central_bank
			Not = {
                has_modifier = global_monetary_reference
            }
		}
		add_modifier = {
			name = strong_currency
			multiplier = difference_with_average_gold_exchange_rate_currencies_revers_100 
		}
	}
	#weak_currency
	if = {
		limit = {
			has_modifier = weak_currency
		}
		remove_modifier = weak_currency
	}
	if = {
		limit = {
			money_value_rapported_inflation < global_var:money_value_median
			money_value_rapported_inflation > 0.0001
			has_modifier = has_central_bank
		}
		add_modifier = {
			name = weak_currency
			multiplier = difference_with_average_gold_exchange_rate_currencies_revers_100
		}

	}
	#extrem_weak_currency
	if = {
		limit = {
			has_modifier = extreme_weak_currency
		}
		remove_modifier = extreme_weak_currency
	}
	if = {
		limit = {
			money_value_rapported_inflation < 0.001
			has_modifier = has_central_bank
		}
		add_modifier = {
			name = extreme_weak_currency
			multiplier = difference_with_average_gold_exchange_rate_currencies_revers_100
		}

	}
}
inflation_modifier = {
	if = {
		limit = {
			has_modifier = inflation_country
		}
		remove_modifier = inflation_country
	}
	if = {
		limit = {
			has_modifier = deflation_country
		}
		remove_modifier = deflation_country
	}
	if = {
		limit = {
			has_modifier = foreign_exchange_controls
			not = {
				or = {
					var:devaluation_on = 1
					var:revaluation_on = 1
				}
			}
		}
		remove_modifier = foreign_exchange_controls
	}
	if = {
		limit = {
 			#has_law = law_type:law_revaluation
		 	inflation_value > 0.02
		}
		add_modifier = {
			name = inflation_country
			multiplier = inflation_value
		}
	}
	else_if = {
		limit = {
			#has_law = law_type:law_devaluation
			inflation_value < -0.02
		}
		add_modifier = {
			name = deflation_country
			multiplier = inflation_value
		}
	}
	if = {
		limit = {
 			inflation_value > 0.5
			not = {
				or = {
					var:devaluation_on = 1
					var:revaluation_on = 1
				}
			}
		}
		add_modifier = {
			name = foreign_exchange_controls
		}
	}
	if = {
		limit = {
 			inflation_value < -0.5
			not = {
				or = {
					var:devaluation_on = 1
					var:revaluation_on = 1
				}
			}
		}
		add_modifier = {
			name = foreign_exchange_controls
		}
	}
}
