economic_sentiment_index_base_1_button = { #5 PCS aléatoire /6mois
	name = "economic_sentiment_index_base_1_button"
	desc = "economic_sentiment_index_base_1_button_desc"

	possible = {  
		NOT = {
			has_modifier = economic_sentiment_index_modifier_1 
		}
		custom_tooltip = {
			text = economic_sentiment_index_base_1_button_tt
			economic_sentiment_index_base_JE_fixe > 1
		} 
		is_player = yes
	}

	effect = { 
		if = {
			limit  = {
				building_ef_private_construction_lvl_to_build > 0
				has_modifier = has_financial_center
			} 
			while = {
				count = 5 #pour etre sure de tout prendre à voir en fin de partie 
				
				random_scope_state = {
					limit = {
						not = {
							building_ef_private_construction_lvl_to_build = 0
						}
						building_urban_center_lvl_by_base_rate >= building_ef_private_construction_lvl_state
					} 
					start_privately_funded_building_construction = building_ef_private_construction
				}
			} 
		}  
		add_modifier = {
			name = economic_sentiment_index_modifier_1
			months = 12
		} 
	}
}
