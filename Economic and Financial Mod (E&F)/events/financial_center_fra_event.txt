namespace = financial_center_fra_event

financial_center_fra_event.1 = {

	type = country_event
	placement = root
	hidden = yes

	title = "Financial Center I"

	duration = 1

	trigger = {
	}

	immediate = {
		capital = {
			save_scope_as = central_bank_site
		}
		scope:central_bank_site = {
			create_building = {
				building = building_financial_centre_fra
				level = 25
			}
		}
	}
}
