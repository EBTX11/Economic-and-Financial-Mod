speculative_share_1_button = { 
	name = "speculative_share_1_button"
	desc = "speculative_share_1_button_desc"
 
	visible = {
		var:speculative_share_1 > 20
	}
	possible = {  
		NOT = {
			has_modifier = speculative_share_modifier_1 
		}
		custom_tooltip = {
			text = speculative_share_1_button_tt
			#var:speculative_share_1 > 40
		} 
		is_player = yes
	}
	effect = {      
		add_banned_goods = g:bond
		add_banned_goods = g:manufacture_stock
		add_banned_goods = g:agricultural_stock
		add_banned_goods = g:mining_stock
		add_banned_goods = g:railroad_stock 
		 
		remove_modifier = speculative_share_modifier_1
		add_modifier = {
			name = speculative_share_modifier_1
			months = 12
		}  
	}
}
