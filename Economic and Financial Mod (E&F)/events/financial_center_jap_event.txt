namespace = financial_center_jap_event

financial_center_jap_event.1 = {

	type = country_event
	placement = root
	hidden = yes

	title = "Financial Center I"

	duration = 1

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_KANTO
			}
			save_scope_as = financial_center_state
		}
		scope:financial_center_state = {
			create_building = {
				building = building_financial_centre_jap
				level = 50
			}
		}
	}
}
