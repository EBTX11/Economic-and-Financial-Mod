bank_je_central_1 = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	is_shown_when_inactive = {
		has_technology_researched = urbanization
	}
	
	complete = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		has_technology_researched = international_trade
		is_revolutionary = no
		gdp > 10000000
		game_date >= 1836.1.15
		NOT = { 
			has_law = law_shares_currency
			has_law = law_uk_currency
			has_law = law_peso_currency
			has_law = law_por_currency
			has_law = law_brz_currency
			has_law = law_fra_currency
			has_law = law_usa_currency
			has_law = law_germany_currency
			has_law = law_aus_currency
			has_law = law_chi_currency
			has_law = law_bic_currency
			has_law = law_jap_currency
			has_law = law_spa_currency
			has_law = law_tur_currency
			has_law = law_rus_currency
			has_law = law_ita_currency
			has_law = law_net_currency
			has_law = law_bel_currency
			has_law = law_swi_currency
			has_law = law_dinar_currency
			has_law = law_krone_currency
			has_law = law_leu_currency
			has_law = law_drachma_currency
			has_law = law_qiran_currency
			has_law = law_bath_currency
		}
	}
	should_be_pinned_by_default = yes

	on_complete = {
		if = {
			trigger_event = { 
				id = bank_event_central.1
			}
		}
	}
}
