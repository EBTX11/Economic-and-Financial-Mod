namespace = bank_event_central

bank_event_central.1 = {
	type = country_event
	placement = scope:central_bank_site
	hidden = yes

	title = "Central Bank I"

	duration = 1

	trigger = {
	}

	immediate = {
		capital = {
			save_scope_as = central_bank_site
		}
		scope:central_bank_site = {
			create_building = {
				building = building_bank
				level = 5
			}
		}
	}
}
