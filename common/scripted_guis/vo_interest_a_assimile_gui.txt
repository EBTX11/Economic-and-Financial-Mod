vo_lunch_national_debt = {
	effect = {
		add_modifier = {
			name = lunch_national_debt
			multiplier = national_debt_repay_monthly
			years = 10
		}
	}
}

vo_increase_interest = {
	is_shown = {
		vo_interest_value <= 0.10
	}
	effect = {
		if = {
			limit = {
				has_variable = bank_interest
			}
			change_variable = {
				name = bank_interest
				add = 0.001
			}
		}
		else = {
			set_variable = {
				name = bank_interest
				value = 0.001
			}
		}
		remove_modifier = central_bank_interest
		add_modifier = {
			name = central_bank_interest
			multiplier = vo_interest_value
		}
	}
}
vo_reduce_interest = {
	is_shown = {
		vo_interest_value >= 0.001
	}
	effect = {

				subtract = 0.001
			}
		}
		else = {
			set_variable = {
				name = bank_interest
				subtract = 0.001
			}
		}
		remove_modifier = central_bank_interest
		add_modifier = {
			name = central_bank_interest
			multiplier = vo_interest_value
		}
	}
}
