namespace = unique_companies_events

# Unique companies search for Countries without Banks
unique_companies_events.1 = {
	type = country_event
	hidden = yes

	trigger = {
		any_country = {
			is_player = no
			has_technology_researched = investment_banks
			unique_companies_has_bank = no
		}
	}

	immediate = {
		trigger_event = { id = unique_companies_events.2 days = 0 popup = no }
	}
}
