law_event_peso_costa_rican_colon.1 = {
	type = country_event
	placement = root
	hidden = yes

	title = "Current circulation - peso_costa_rican_colon"

	duration = 1

	immediate = {                                     
		capital = {
			save_scope_as = central_bank_site
		}
		
		set_variable = {
			name = money_value_target_1
			value = { gold_to_silver_rate gold_to_silver_rate_x_2 }
		}
		change_variable = {
			name = circulating_peso_costa_rican_colon_c_var_1
			add = { 50000009 100000009 }
		}
		scope:central_bank_site = {
			change_variable = {
				name = stockpiling_peso_costa_rican_colon_state_1
				add  = {
					value = owner.var:circulating_peso_costa_rican_colon_c_var_1
					multiply = 0.1
				}
			}
			change_variable = {
				name = silver_state_1
				add = {
					value = owner.var:circulating_peso_costa_rican_colon_c_var_1
					add = var:stockpiling_peso_costa_rican_colon_state_1
					multiply = owner.var:money_value_target_1
				}
			}
			change_variable = {
				name = gold_state_1
				add = {
					value = var:silver_state_1
					multiply = silver_to_gold_rate
				}
			}
		}
		activate_law = law_type:law_peso_costa_rican_colon_currency
		if = {
			limit = {
				not = { has_law = law_type:law_peso_costa_rican_colon_currency_2 }
			}
			trigger_event = { id = law_event_peso_costa_rican_colon.3 }
		}
	}
}
